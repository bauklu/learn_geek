1.Написать скрипт, который удаляет из текстового файла пустые строки и 
заменяет маленькие символы на большие. 


lb@lb-VirtualBox:~$ cat file15

__________________________________________________

#!/bin/bash

#удаляет из текстового файла пустые строки

sed '/^[[:space:]]*$/d' file15 | tr [:lower:] [:upper:]

#и заменяет маленькие символы на большие

___________________________________________________

lb@lb-VirtualBox:~$ vim script.sh
lb@lb-VirtualBox:~$ ./script.sh
